class Solution {
public:
    vector<string> fizzBuzz(int n) {
        vector<string>res;
        int i=0;
        
        while(i++<n)
        {
           if(i%5==0 && i%3==0)
               res.push_back("FizzBuzz");
           else if(i%3==0)
               res.push_back("Fizz");
            else if(i%5==0)
                res.push_back("Buzz");
            else
                res.push_back(to_string(i));
        }
        
        return res;
        
    }
};
